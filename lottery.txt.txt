<script>
    function generateLottery() {
        const totalNumbers = 42;
        const pickCount = 6;
        const suggestionsCount = 4;
        
        let allPicked = [];
        let htmlOutput = "";

        for (let i = 1; i <= suggestionsCount; i++) {
            let pool = Array.from({length: totalNumbers}, (_, i) => i + 1);
            let selection = [];

            // Equivalent to random.sample: pick unique numbers
            for (let j = 0; j < pickCount; j++) {
                const randomIndex = Math.floor(Math.random() * pool.length);
                const pickedNumber = pool.splice(randomIndex, 1)[0];
                selection.push(pickedNumber);
                allPicked.push(pickedNumber);
            }

            // Sort numerically
            selection.sort((a, b) => a - b);
            htmlOutput += `<div class="suggestion"><strong>#${i}:</strong> ${selection.join(", ")}</div>`;
        }

        // Display Suggestions
        document.getElementById('output').innerHTML = htmlOutput;

        // Calculate Frequencies (The "Repeated" part)
        const counts = {};
        allPicked.forEach(num => { counts[num] = (counts[num] || 0) + 1; });

        // Display Frequencies
        let statsHtml = "<strong>Frequency:</strong><br>";
        Object.keys(counts).sort((a, b) => a - b).forEach(num => {
            const repeatText = counts[num] > 1 ? `<span class="highlight">(${counts[num]}x)</span>` : `(${counts[num]}x)`;
            statsHtml += `Num ${num}: ${repeatText}<br>`;
        });
        document.getElementById('stats').innerHTML = statsHtml;
    }
</script>
